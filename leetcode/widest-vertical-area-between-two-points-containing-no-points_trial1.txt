class Solution:
    def maxWidthOfVerticalArea(self, points: List[List[int]]) -> int:
        points.sort()
        max_area = 0
        for i in range(len(points) - 1):
            if abs(points[i][0] - points[i+1][0]) > max_area:
                max_area = abs(points[i][0] - points[i+1][0])

        return max_area
